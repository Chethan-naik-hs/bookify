// cart.js document.addEventListener("DOMContentLoaded", function () { // Function to get the cart from sessionStorage function getCart() { return JSON.parse(sessionStorage.getItem("cart")) || []; } // Function to update the cart in sessionStorage function updateCart(cart) { sessionStorage.setItem("cart", JSON.stringify(cart)); } // Function to update the total price displayed in the cart function updateTotal() { const cart = getCart(); const total = cart.reduce((sum, item) => sum + parseFloat(item.price), 0).toFixed(2); document.getElementById("total").textContent = total; } // Function to add a product to the cart function addToCart(product) { // Check if the price is greater than 0 if (parseFloat(product.price) > 0) { const cart = getCart(); cart.push(product); updateCart(cart); updateTotal(); } else { // Display an error message alert("Cannot add a product with a price of 0.00 to the cart."); } } // Display existing items in the cart on page load const cartItemsContainer = document.querySelector(".cart-items-container"); const cart = getCart(); cart.forEach(product => { const cartItem = document.createElement("div"); cartItem.textContent = `${product.name} - â‚¹${parseFloat(product.price).toFixed(2)}`; cartItemsContainer.appendChild(cartItem); }); // Get the "Add to Cart" button from the product details page const addToCartButton = document.getElementById("add-to-cart"); // Handle "Add to Cart" button click addToCartButton.addEventListener("click", () => { const product = getQueryParams(); // Assuming getQueryParams() is defined in script.js or a relevant file addToCart(product); }); });
